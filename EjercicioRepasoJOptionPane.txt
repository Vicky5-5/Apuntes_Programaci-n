package PracticasPruebas;

import javax.swing.JOptionPane;

public class CalcularPeso {
	public static void main(String[] args) {
//Calcular el peso si eres  hombre (altura en cm - 110 y en las mujeres lo mismo pero restando 120)
		
		//con jpotionpane
		
		String insertar = JOptionPane.showInputDialog("Inserta tu sexo. Si eres hombre H y si eres mujer F");
		if(insertar.equalsIgnoreCase("h")) {
			String alturaStr = JOptionPane.showInputDialog("Introduce tu altura en cm:");
            int altura = Integer.parseInt(alturaStr);
            int pesoIdeal = altura - 110;
            JOptionPane.showMessageDialog(null, "Tu peso ideal es: " + pesoIdeal + " kg");
		} else if(insertar.equalsIgnoreCase("m")) {
			String alturaStr = JOptionPane.showInputDialog("Introduce tu altura en cm:");
            int altura = Integer.parseInt(alturaStr);
            int pesoIdeal = altura - 120;
            JOptionPane.showMessageDialog(null, "Tu peso ideal es: " + pesoIdeal + " kg");
		}
	
	}
}
