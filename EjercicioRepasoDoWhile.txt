		//con jpotionpane
		
		String genero="";
		do {
			genero = JOptionPane.showInputDialog("Introduce tu g√©nero (H/M)");

		}while(genero.equalsIgnoreCase("h") == false && genero.equalsIgnoreCase("m")== false);
